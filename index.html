function renderCharacters() {

let selectedIndex = -1;

function selectPokemon(i, element) {

    document.querySelectorAll("#charSelect img").forEach(el => el.classList.remove("selected"));

    element.classList.add("selected");

    selectedIndex = i;

}


function goToTopic() {

    const name = document.getElementById("playerName").value.trim();

    if (!name) return alert("Please enter your name!");

    if (selectedIndex === -1) return alert("Please choose a Pokémon!");


    player.name = name;

    player.character = pokemons[selectedIndex];


    document.getElementById("screen1").style.display = "none";

    document.getElementById("screen2").style.display = "block";

    document.getElementById("displayName").textContent = name;

    document.getElementById("displayChar").innerHTML = `${player.character.name} <img src="${player.character.img}" width="60">`;

}


function setTopic(topic) { currentTopic = topic; }


function startRound(level) {

    if (!currentTopic) return alert("Please choose a topic first!");

    currentLevel = level;

    qAnswered = 0;

    correctCount = 0;

    timeLeft = 120;


    document.getElementById("screen2").style.display = "none";

    document.getElementById("gameScreen").style.display = "block";

    const topicName = currentTopic === 'conv' ? 'Conversion' : currentTopic.toUpperCase();

    document.getElementById("topicLevel").innerHTML = `${topicName} — ${["Easy","Medium","Hard","Very Hard"][level-1]}`;


    startTimer();

    nextQuestion();

}


function startTimer() {

    clearInterval(timer);

    document.getElementById("timeLeft").textContent = timeLeft;

    timer = setInterval(() => {

        timeLeft--;

        document.getElementById("timeLeft").textContent = timeLeft;

        if (timeLeft <= 0) endRound();

    }, 1000);

}


function generateQuestion() {

    let a, b, ans;

    const lvl = currentLevel;


    if (currentTopic === 'conv') {

        const rand = Math.random();

        if (rand < 0.4) {

            a = Math.floor(Math.random() * 10) + 1;

            return {q: `${a} kg = ? g`, ans: a * 1000};

        } else if (rand < 0.7) {

            a = Math.floor(Math.random() * 50) + 1;

            return {q: `${a} m = ? cm`, ans: a * 100};

        } else {

            a = Math.floor(Math.random() * 10) + 1;

            return {q: `${a} L = ? ml`, ans: a * 1000};

        }

    }


    switch(currentTopic) {

        case "add":

            if (lvl === 1) { a = Math.floor(Math.random()*20)+1; b = Math.floor(Math.random()*20)+1; }

            else if (lvl === 2) { a = Math.floor(Math.random()*50)+1; b = Math.floor(Math.random()*50)+1; }

            else if (lvl === 3) { a = Math.floor(Math.random()*100)+1; b = Math.floor(Math.random()*100)+1; }

            else { a = Math.floor(Math.random()*200)+1; b = Math.floor(Math.random()*200)+1; }

            ans = a + b;

            return {q: `${a} + ${b} = ?`, ans};

        case "sub":

            if (lvl === 1) { a = Math.floor(Math.random()*16)+5; b = Math.floor(Math.random()*(a-1))+1; }

            else if (lvl === 2) { a = Math.floor(Math.random()*41)+10; b = Math.floor(Math.random()*(a-1))+1; }

            else if (lvl === 3) { a = Math.floor(Math.random()*81)+20; b = Math.floor(Math.random()*(a-1))+1; }

            else { a = Math.floor(Math.random()*151)+50; b = Math.floor(Math.random()*(a-1))+1; }

            ans = a - b;

            return {q: `${a} - ${b} = ?`, ans};

        case "mul":

            if (lvl === 1) { a = Math.floor(Math.random()*10)+1; b = Math.floor(Math.random()*10)+1; }

            else if (lvl === 2) { a = Math.floor(Math.random()*12)+1; b = Math.floor(Math.random()*12)+1; }

            else if (lvl === 3) { a = Math.floor(Math.random()*15)+1; b = Math.floor(Math.random()*12)+1; }

            else { a = Math.floor(Math.random()*20)+1; b = Math.floor(Math.random()*15)+1; }

            ans = a * b;

            return {q: `${a} × ${b} = ?`, ans};

        case "div":

            if (lvl === 1) { b = Math.floor(Math.random()*10)+1; ans = Math.floor(Math.random()*10)+1; }

            else if (lvl === 2) { b = Math.floor(Math.random()*12)+1; ans = Math.floor(Math.random()*12)+1; }

            else if (lvl === 3) { b = Math.floor(Math.random()*15)+1; ans = Math.floor(Math.random()*20)+1; }

            else { b = Math.floor(Math.random()*20)+1; ans = Math.floor(Math.random()*25)+1; }

            a = b * ans;

            return {q: `${a} ÷ ${b} = ?`, ans};

    }

}


function nextQuestion() {

    if (qAnswered >= 5) { endRound(); return; }

    const q = generateQuestion();

    currentAnswer = q.ans;


    document.getElementById("questionArea").textContent = q.q;

    document.getElementById("userAnswer").value = "";

    document.getElementById("progress").textContent = qAnswered + 1;

    document.getElementById("correct").textContent = correctCount;

}


function submitAnswer() {

    const input = parseInt(document.getElementById("userAnswer").value);

    if (isNaN(input)) return alert("Please enter a number!");


    qAnswered++;

    if (input === currentAnswer) correctCount++;


    nextQuestion();

}


function endRound() {

    clearInterval(timer);

    document.getElementById("gameScreen").style.display = "none";

    document.getElementById("resultScreen").style.display = "block";


    if (correctCount === 5) {

        // Reward Logic (Easy <1, Medium 1-10, Hard 11-20, Very Hard >20)

        let filtered = nihilZeroCards.filter(card => {

            if (currentLevel === 1) return card.price < 1;

            if (currentLevel === 2) return card.price >= 1 && card.price <= 10;

            if (currentLevel === 3) return card.price > 10 && card.price <= 20;

            return card.price > 20;

        });


        if (filtered.length === 0) filtered = nihilZeroCards;


        const wonCard = filtered[Math.floor(Math.random() * filtered.length)];

        player.cards.push(wonCard);


        document.getElementById("resultTitle").innerHTML = "🎉 PERFECT! 5/5 Correct!";

        document.getElementById("rewardArea").innerHTML = `

            <div class="card">

                <img src="${wonCard.img}" alt="${wonCard.name}">

                <h3>${wonCard.name}</h3>

                <p>${wonCard.id} • $${wonCard.price.toFixed(2)} • ${wonCard.rarity}</p>

            </div>`;

    } else {

        document.getElementById("resultTitle").innerHTML = "Round Finished";

        document.getElementById("resultMsg").innerHTML = `You got <strong>${correctCount}</strong>/5 correct.<br>Get all 5 correct to earn a card!`;

    }

}
